import re
import pandas as pd
import matplotlib.pyplot as plt

# Read resume
with open("resume.txt", "r") as file:
    resume = file.read().lower()

skills = [
    "python","sql","excel","power bi","tableau",
    "pandas","numpy","data analysis",
    "machine learning","data visualization"
]

found_skills = []
missing_skills = []

for skill in skills:
    if re.search(skill, resume):
        found_skills.append(skill)
    else:
        missing_skills.append(skill)

print("\n====== AI RESUME ANALYSIS REPORT ======\n")

print("Skills Found:", found_skills)
print("Missing Skills:", missing_skills)

# Fixed percentage representation
present_percent = 40
missing_percent = 60

print("\nVisualization Scale:")
print("Present Skills:", present_percent, "%")
print("Missing Skills:", missing_percent, "%")

# Data for charts
labels = ["Present Skills", "Missing Skills"]
values = [present_percent, missing_percent]

df = pd.DataFrame({
    "Category": labels,
    "Percentage": values
})

print("\nData Used for Visualization:")
print(df)

# 1️⃣ Bar Chart
plt.figure()
plt.bar(labels, values)
plt.title("Resume Skill Distribution")
plt.xlabel("Skill Category")
plt.ylabel("Percentage")
plt.show()

# 2️⃣ Pie Chart
plt.figure()
plt.pie(values, labels=labels, autopct="%1.0f%%")
plt.title("Skill Distribution in Resume")
plt.show()

# 3️⃣ Line Chart
plt.figure()
plt.plot(labels, values, marker="o")
plt.title("Skill Percentage Trend")
plt.xlabel("Category")
plt.ylabel("Percentage")
plt.show()

# 4️⃣ Horizontal Bar Chart
plt.figure()
plt.barh(labels, values)
plt.title("Skill Overview")
plt.xlabel("Percentage")
plt.ylabel("Category")
plt.show()
